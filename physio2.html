Connect to Supabase

Quick Setup Guide:

  1. Go to supabase.com and create a new project
  2. Find your project URL (format: https://abc123.supabase.co)
  3. Get your anon key from Settings API
  4. Run the SQL commands from the setup instructions to create tables

Format: https://[project-ref].supabase.co

Find in Settings API anon/public key

Database Setup Required:

After connecting, run these SQL commands in your Supabase SQL Editor:

-- IMPORTANT: Run these SQL commands in your Supabase SQL Editor
-- Go to: Your Supabase Dashboard  SQL Editor  New Query

-- 1. Create patients table
CREATE TABLE IF NOT EXISTS patients (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  first_name TEXT NOT NULL,
  last_name TEXT NOT NULL,
  email TEXT,
  phone TEXT,
  date_of_birth DATE,
  address TEXT,
  medical_history TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 2. Create physiotherapists table (REQUIRED for staff management)
CREATE TABLE IF NOT EXISTS physiotherapists (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  first_name TEXT NOT NULL,
  last_name TEXT NOT NULL,
  email TEXT UNIQUE NOT NULL,
  phone TEXT,
  license_number TEXT UNIQUE NOT NULL,
  years_experience INTEGER DEFAULT 0,
  specializations TEXT[], -- Array of specializations
  employment_status TEXT DEFAULT 'full-time',
  hourly_rate DECIMAL(10,2),
  start_date DATE,
  bio TEXT,
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 3. Create appointments table  
CREATE TABLE IF NOT EXISTS appointments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  patient_id UUID REFERENCES patients(id),
  physiotherapist_id UUID REFERENCES physiotherapists(id),
  appointment_date TIMESTAMP,
  duration INTEGER,
  status TEXT DEFAULT 'scheduled',
  notes TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 4. Create treatments table
CREATE TABLE IF NOT EXISTS treatments (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  patient_id UUID REFERENCES patients(id),
  appointment_id UUID REFERENCES appointments(id),
  physiotherapist_id UUID REFERENCES physiotherapists(id),
  treatment_type TEXT,
  description TEXT,
  notes TEXT,
  duration INTEGER DEFAULT 60, -- Duration in minutes
  treatment_date TIMESTAMP DEFAULT NOW(),
  created_at TIMESTAMP DEFAULT NOW()
);

-- 5. Create exercises table
CREATE TABLE IF NOT EXISTS exercises (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  category TEXT,
  difficulty_level TEXT,
  instructions TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 6. DISABLE Row Level Security for easier development (IMPORTANT!)
-- Note: For production, you should enable RLS and create proper policies
ALTER TABLE patients DISABLE ROW LEVEL SECURITY;
ALTER TABLE physiotherapists DISABLE ROW LEVEL SECURITY;
ALTER TABLE appointments DISABLE ROW LEVEL SECURITY;
ALTER TABLE treatments DISABLE ROW LEVEL SECURITY;
ALTER TABLE exercises DISABLE ROW LEVEL SECURITY;
